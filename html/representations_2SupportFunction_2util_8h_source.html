42  const std::pair<matrix_t<Number>, vector_t<Number>>& getParameterSet(unsigned exponent) const {
43  //std::cout << this << " Request parameter set for exponent " << exponent << std::endl;
44  if(parameters.find(exponent) != parameters.end()){
45  return parameters.at(exponent);
46  }
47  // create new parameters
48  while((--parameters.end())->first < exponent) {
49  createNextReduct();
50  }
51  assert(parameters.find(exponent) != parameters.end());
52  return parameters.at(exponent);
53  };

55  void createNextReduct() const {
56  // the last created reduction pair is at the back, as a std::map is per default sorted ascending
57  assert(parameters.size() > 0);
58  std::size_t powerOfTwo = carl::pow(2, power);
59  // first compute the new b
60  vector_t<Number> bTrans = (--parameters.end())->second.second;
61  matrix_t<Number> aTrans = (--parameters.end())->second.first;
62  //std::cout << "Starting from exponent " << (--matrices.end())->first << std::endl;
63  unsigned exponent = ((--parameters.end())->first) * powerOfTwo;
64  //std::cout << "New exponent: " << exponent << std::endl;
65  // accumulate b
66  for (std::size_t i = 1; i < powerOfTwo ; i++){
67  // Note: aTrans hasn't changed yet -> we can use it for transformation.
68  bTrans = aTrans*bTrans + (--parameters.end())->second.second;
69  }
70  // accumulate A
71  for (std::size_t i = 0; i < power; i++){
72  aTrans = aTrans*aTrans;
73  }
74  parameters[exponent] = std::make_pair(aTrans, bTrans);
75  }
