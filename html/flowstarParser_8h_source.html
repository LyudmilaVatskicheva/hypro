112  void addContinuousState(const std::vector<matrix_t<Number>>& _constraint, unsigned id, std::vector<RawState<Number>>& _states) {
113  if(_constraint.size() > 0) {
114  //std::cout << "Add continuous state for location " << id << std::endl;
115  bool found = false;
116  // due to the creation of initial states, which uses this function as well and the occurrence of duplicates in the initial states, we need the last instance of a matching state.
117  for(auto stateIt = _states.rbegin(); stateIt != _states.rend(); ++stateIt) {
118  if(stateIt->location->id() == id){
119  //std::cout << "stateIt->already exists." << std::endl;
120  found = true;
121  assert(stateIt->discreteAssignment.size() == mDiscreteVariableIds.size());
122  unsigned constraintsNum = stateIt->set.first.rows();
123  unsigned dimension = constraintsNum > 0 ? stateIt->set.first.cols() : (_constraint.begin()->cols())-1;
124  //std::cout << "current constraints: " << boost::get<cPair<Number>>(stateIt->set).first << std::endl;
125  cPair<Number> set = stateIt->set;
126  //std::cout << "Resize to " << constraintsNum+_constraint.size() << " x " << dimension << std::endl;
127  set.first.conservativeResize(constraintsNum+_constraint.size(), dimension);
128  set.second.conservativeResize(constraintsNum+_constraint.size());
129  //std::cout << "State already existing with " << constraintsNum << " rows." << std::endl;
130  for(const auto& row : _constraint) {
131  assert(row.rows() == 1);
132  assert(row.cols() > 0);
133  //std::cout << "add row " << constraintsNum;
134  //std::cout << "row properties: " << row.rows() << " , " << row.cols() << std::endl;
135  //std::cout << "set properties: " << set.first.rows() << " , " << set.first.cols() << std::endl;
136  set.first.row(constraintsNum) = row.block(0,0,1,row.cols()-1);
137  //std::cout << " " << convert<Number,double>(set.first.row(constraintsNum)) << " <= ";
138  set.second(constraintsNum) = -row(0,row.cols()-1);
139  //std::cout << set.second(constraintsNum) << std::endl;
140  ++constraintsNum;
141  }
142  stateIt->set = set;
143  //std::cout << "New constraints: " << boost::get<cPair<Number>>(stateIt->set).first << std::endl;
144  break;
145  }
146  }
147  if(!found){
148  RawState<Number> s;
149  s.location = mLocationManager.location(id);
150  std::pair<matrix_t<Number>, vector_t<Number>> set;
151  set.first = matrix_t<Number>(_constraint.size(), _constraint.begin()->cols()-1);
152  set.second = vector_t<Number>(_constraint.size());
153  unsigned constraintsNum = 0;
154  //std::cout << "State newly created with " << _constraint.size() << " rows." << std::endl;
155  for(const auto& row : _constraint) {
156  assert(row.rows() == 1);
157  //std::cout << "add row " << constraintsNum;
158  set.first.row(constraintsNum) = row.block(0,0,1,row.cols()-1);
159  //std::cout << " " << set.first.row(constraintsNum) << " <= ";
160  set.second(constraintsNum) = -row(0,row.cols()-1);
161  //std::cout << set.second(constraintsNum) << std::endl;
162  ++constraintsNum;
163  }
164  s.set = set;
165  // initial setup of the discrete assignment
166  for(const auto& id : mDiscreteVariableIds ){
167  s.discreteAssignment[VariablePool::getInstance().carlVarByIndex(id)] = carl::Interval<Number>::unboundedInterval();
168  }
169  _states.emplace_back(s);
170  }
171  }

174  void addDiscreteState(const std::pair<unsigned, carl::Interval<Number>>& _initPair, unsigned id, std::vector<RawState<Number>>& _states) {
175  //std::cout << "Add discrete state for location " << id << std::endl;
176  bool found = false;
177  // due to the creation of initial states, which uses this function as well and the occurrence of duplicates in the initial states, we need the last instance of a matching state.
178  for(auto stateIt = _states.rbegin(); stateIt != _states.rend(); ++stateIt) {
179  assert(stateIt->discreteAssignment.size() == mDiscreteVariableIds.size());
180  if(stateIt->location == mLocationManager.location(id)){
181  found = true;
182  stateIt->discreteAssignment[VariablePool::getInstance().carlVarByIndex(_initPair.first)] = _initPair.second;
183  }
184  assert(stateIt->discreteAssignment.size() == mDiscreteVariableIds.size());
185  break;
186  }
187  if(!found){
188  RawState<Number> s;
189  s.location = mLocationManager.location(id);
190  // initialize all discrete assignments to unbounded, when a new state is created
191  for(const auto& id : mDiscreteVariableIds ){
192  if(id == _initPair.first){
193  s.discreteAssignment[VariablePool::getInstance().carlVarByIndex(_initPair.first)] = _initPair.second;
194  } else {
195  s.discreteAssignment[VariablePool::getInstance().carlVarByIndex(id)] = carl::Interval<Number>::unboundedInterval();
196  }
197  }
198  assert(s.discreteAssignment.size() == mDiscreteVariableIds.size());
199  _states.emplace_back(s);
