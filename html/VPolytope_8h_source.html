210  void setNeighbors( const Point<Number>& _point, const std::set<Point<Number>>& _neighbors ) {
211  unsigned pos = 0;
212  while ( pos < mVertices.size() && ( mVertices[pos] != _point ) ) ++pos;
213 
214  if ( mVertices[pos] == _point ) {
215  for ( const auto& neighbor : _neighbors ) {
216  unsigned nPos = 0;
217  while ( nPos < mVertices.size() && ( mVertices[nPos] != neighbor ) ) ++nPos;
218 
219  if ( mVertices[nPos] == neighbor ) {
220  mNeighbors[pos].insert( nPos );
221  mNeighbors[nPos].insert( pos );
222  }
223  }
224  }

281  template<typename N = Number, carl::DisableIf< carl::is_float<N> > = carl::dummy>
282  void reduceNumberRepresentation(unsigned limit = fReach_DENOMINATOR) const {
283  if(!mVertices.empty()) {
284  // determine barycenter to set rounding directions
285  unsigned dimension = mVertices.begin()->rawCoordinates().rows();
286  vector_t<Number> barycenter = vector_t<Number>::Zero(dimension);
287  for(const auto& vertex : mVertices) {
288  barycenter = barycenter + (vertex.rawCoordinates()/ Number(unsigned(mVertices.size())));
289  }
290  for(auto& vertex : mVertices) {
291  // distance vector to the barycenter - each component now determines its own rounding direction.
292  vector_t<Number> roundingDirections = vertex.rawCoordinates() - barycenter;
293  for(unsigned d = 0; d < dimension; ++d) {
294  assert(d < vertex.dimension());
295  // determine, which is larger: numerator or denominator
296  Number denom = carl::getDenom(vertex.at(d));
297  Number num = carl::getNum(vertex.at(d));
298  if(denom > limit && carl::abs(num) > limit) {
299  if(denom > num) {
300  Number newNum = num/denom*limit;
301  if(roundingDirections(d) >= 0 ) { // round up
302  newNum = carl::ceil(newNum);
303  assert(newNum/limit >= vertex.at(d));
304  } else { // round down
305  newNum = carl::floor(newNum);
306  assert(newNum/limit <= vertex.at(d));
307  }
308  vertex[d] = newNum/limit;
309  } else { // num >= denom
310  Number newDenom = denom/num*limit;
311  if(roundingDirections(d) >= 0 ) { // round up
312  newDenom = carl::floor(newDenom);
313  assert(limit/newDenom >= vertex.at(d));
314  } else { // round down
315  newDenom = carl::ceil(newDenom);
316  assert(limit/newDenom <= vertex.at(d));
317  }
318  vertex[d] = limit/newDenom;
319  }
320  }
321  }
322  }
323  }
324  }

349  static bool insidePlanes(const vector_t<double>& vertex, const matrix_t<double>& normals, const vector_t<double>& offsets) {
350  for(unsigned rowIndex = 0; rowIndex < normals.rows(); ++rowIndex){
351  // compare with tolerance of 128 ULPs.
352  if( !carl::AlmostEqual2sComplement(vertex.dot(normals.row(rowIndex)),offsets(rowIndex), 128) ){
353  DEBUG("hypro.representations.vpolytope","Values " << vertex.dot(normals.row(rowIndex)) << " and " << offsets(rowIndex) << " are not equal.");
354  return false;
