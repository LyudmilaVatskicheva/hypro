181  template<typename N = Number, carl::DisableIf< carl::is_float<N> > = carl::dummy>
182  void reduceNumberRepresentation(unsigned limit = fReach_DENOMINATOR) const {
183  if(!mVertices.empty()) {
184  // determine barycenter to set rounding directions
185  unsigned dimension = mVertices.begin()->rawCoordinates().rows();
186  vector_t<Number> barycenter = vector_t<Number>::Zero(dimension);
187  for(const auto& vertex : mVertices) {
188  barycenter = barycenter + (vertex.rawCoordinates()/ Number(unsigned(mVertices.size())));
189  }
190  for(auto& vertex : mVertices) {
191  // distance vector to the barycenter - each component now determines its own rounding direction.
192  vector_t<Number> roundingDirections = vertex.rawCoordinates() - barycenter;
193  for(unsigned d = 0; d < dimension; ++d) {
194  assert(d < vertex.dimension());
195  // determine, which is larger: numerator or denominator
196  Number denom = carl::getDenom(vertex.at(d));
197  Number num = carl::getNum(vertex.at(d));
198  if(denom > limit && carl::abs(num) > limit) {
199  if(denom > num) {
200  Number newNum = num/denom*limit;
201  if(roundingDirections(d) >= 0 ) { // round up
202  newNum = carl::ceil(newNum);
203  assert(newNum/limit >= vertex.at(d));
204  } else { // round down
205  newNum = carl::floor(newNum);
206  assert(newNum/limit <= vertex.at(d));
207  }
208  vertex[d] = newNum/limit;
209  } else { // num >= denom
210  Number newDenom = denom/num*limit;
211  if(roundingDirections(d) >= 0 ) { // round up
212  newDenom = carl::floor(newDenom);
213  assert(limit/newDenom >= vertex.at(d));
214  } else { // round down
215  newDenom = carl::ceil(newDenom);
216  assert(limit/newDenom <= vertex.at(d));
217  }
218  vertex[d] = limit/newDenom;
219  }
220  }
221  }
222  }
223  }
