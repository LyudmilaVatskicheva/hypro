658 int effectiveDimension(const std::vector<Point<Number>>& vertices) {
659  if(!vertices.empty()){
660  if(vertices.size() == 1) {
661  return 0;
662  }
663  unsigned maxDim = vertices.begin()->rawCoordinates().rows();
664  matrix_t<Number> matr = matrix_t<Number>(vertices.size()-1, maxDim);
665  // use first vertex as origin, start at second vertex
666  unsigned rowIndex = 0;
667  for(auto vertexIt = ++vertices.begin(); vertexIt != vertices.end(); ++vertexIt, ++rowIndex) {
668  matr.row(rowIndex) = (vertexIt->rawCoordinates() - vertices.begin()->rawCoordinates()).transpose();
669  }
670  return matr.fullPivLu().rank();
671  }
672  return -1;
673 }
