206 static bool computePostCondition( const hypro::Transition<Number>& _trans, Representation _val,
207  Representation& result ) {
208  // Polytope that is defined by the guard
209  Representation guardPoly = Representation( _trans.guard().mat, _trans.guard().vec );
210 
211  // intersection between valuation polytope and guard polytope
212  Representation intersectionPoly = _val.intersect( guardPoly );
213 
214  // check if the intersection is empty
215  if ( !intersectionPoly.empty() ) {
216  hypro::vector_t<Number> translateVec = _trans.reset().vec;
217  hypro::matrix_t<Number> transformMat = _trans.reset().mat;
218 
219  // perform translation + transformation on intersection polytope
220  result = intersectionPoly.linearTransformation( transformMat, translateVec );
221  return true;
222  } else {
223  return false;
