154 vector_t<Number> computeArithmeticMeanPoint(const std::vector<Point<Number>>& pointVec){
155  //defines an empty solution vector
156  if(!pointVec.empty()){
157  vector_t<Number> res = vector_t<Number>::Zero(pointVec.at(0).rawCoordinates().rows());
158 
159  //computes the arithmetic mean by first building the sum of all points and then dividing it by the number of points
160  for (unsigned i=0; i < pointVec.size(); ++i){
161  assert(res.rows() == pointVec.at(i).rawCoordinates().rows());
162  res += pointVec.at(i).rawCoordinates();
163  }
164  res = res*( ((Number) 1)/pointVec.size());
165  return res;
166  }
167  return vector_t<Number>::Zero(0);
168 }
