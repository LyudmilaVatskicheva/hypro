289  template<typename N = Number, carl::DisableIf< std::is_same<N, double> > = carl::dummy>
290  void reduceNumberRepresentation(const std::vector<Point<Number>>& _vertices = std::vector<Point<Number>>(), unsigned limit = fReach_DENOMINATOR) const {
291  #ifdef REDUCE_NUMBERS
292  #ifdef HPOLY_DEBUG_MSG
293  std::cout << "Attempt to reduce numbers." << std::endl;
294  #endif
295  //std::cout << __func__ << ": Nr. Planes: " << mHPlanes.size() << std::endl;
296  std::vector<Point<Number>> originalVertices;
297  if(_vertices.empty()) {
298  originalVertices = this->vertices();
299  } else {
300  originalVertices = _vertices;
301  }
302 
303  // normal reduction
304  for(unsigned planeIndex = 0; planeIndex < mHPlanes.size(); ++planeIndex){
305  #ifdef HPOLY_DEBUG_MSG
306  std::cout << "Original: " << mHPlanes.at(planeIndex) << std::endl;
307  #endif
308  // find maximal value
309  Number largest = Number(0);
310  mHPlanes.at(planeIndex).makeInteger();
311  #ifdef HPOLY_DEBUG_MSG
312  std::cout << "As Integer: " << mHPlanes.at(planeIndex) << std::endl;
313  #endif
314  largest = carl::abs(mHPlanes.at(planeIndex).offset());
315  for(unsigned i = 0; i < mDimension; ++i){
316  if(carl::abs(mHPlanes.at(planeIndex).normal()(i)) > largest){
317  largest = carl::abs(mHPlanes.at(planeIndex).normal()(i));
318  }
319  }
320 
321  // reduce, if reduction is required
322  if(largest > (limit*limit)) {
323  #ifdef HPOLY_DEBUG_MSG
324  std::cout << "Actual reduction" << std::endl;
325  #endif
326  vector_t<Number> newNormal(mDimension);
327  for(unsigned i = 0; i < mDimension; ++i){
328  newNormal(i) = carl::floor(Number((mHPlanes.at(planeIndex).normal()(i)/largest)*Number(limit)));
329  assert(carl::abs(Number(mHPlanes.at(planeIndex).normal()(i)/largest)) <= Number(1));
330  assert(carl::isInteger(newNormal(i)));
331  assert(newNormal(i) <= Number(limit));
332  }
333  mHPlanes.at(planeIndex).setNormal(newNormal);
334  Number newOffset = mHPlanes.at(planeIndex).offset();
335  newOffset = carl::ceil(Number((newOffset/largest)*Number(limit)));
336  for(const auto& vertex : originalVertices) {
337  Number tmp = newNormal.dot(vertex.rawCoordinates());
338  if(tmp > newOffset){
339  newOffset = newOffset + (tmp-newOffset);
340  assert(newNormal.dot(vertex.rawCoordinates()) <= newOffset);
341  }
342  assert(Halfspace<Number>(newNormal,newOffset).contains(vertex));
343  }
344  newOffset = carl::ceil(newOffset);
345  #ifdef HPOLY_DEBUG_MSG
346  std::cout << "Reduced to " << convert<Number,double>(newNormal).transpose() << " <= " << carl::toDouble(newOffset) << std::endl;
347  #endif
348  mHPlanes.at(planeIndex).setOffset(newOffset);
349  }
350  #ifdef HPOLY_DEBUG_MSG
351  std::cout << "Reduced: " << mHPlanes.at(planeIndex) << std::endl;
352  #endif
353  }
354  #ifdef HPOLY_DEBUG_MSG
355  std::cout << "After Reduction: " << *this << std::endl;
356  #endif
357  #endif
358  }
